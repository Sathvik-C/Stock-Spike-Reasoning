# To run the Backend (FastAPI)
# Open a new terminal tab, then run:

cd backend
source .venv/bin/activate
pip install -r requirements.txt  # Optional, run if dependencies are missing
uvicorn app.main:app --reload --host 0.0.0.0 --port 8000


# To run the Frontend (React)
# Open a second, separate terminal tab, then run:

cd frontend
npm install  # Optional, run if dependencies are missing
npm run dev
